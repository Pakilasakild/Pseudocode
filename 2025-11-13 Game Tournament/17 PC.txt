method Players (PLAYERS, MATCH)
PL = new Array(2)
 VSLOC
 loop I from 0 to MATCH.Length()-1
  if (MATCH[I] == "v" AND MATCH[I+1] == "s") then
   VSLOC = I
  end if
 end loop
 PL[0] = MATCH.SubStr(0, VSLOC-2)
 PL[1] = MATCH.SubStr(VSLOC+3, MATCH.Length()-VSLOC-2)
 return PL
end method
